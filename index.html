// Initialize Icons lucide.createIcons(); function launchSequence() { // 1. Target the inputs const portalInput = document.querySelector('input[placeholder="PORTAL NAME"]'); const statusMsg = document.querySelector('.heavy.text-orange-600.text-xs'); // 2. Perform Validation if (!portalInput.value || portalInput.value.length < 3) { // Error Feedback portalInput.classList.add('border-red-600'); statusMsg.innerText = "ERROR: INVALID PORTAL COORDINATES"; statusMsg.classList.replace('text-orange-600', 'text-red-600'); // Vibration effect for error document.getElementById('form-container').animate([ { transform: 'translateX(-5px)' }, { transform: 'translateX(5px)' } ], { duration: 100, iterations: 3 }); return; } // 3. Successful Validation -> Start Ignition const overlay = document.getElementById('ignition'); overlay.style.display = 'flex'; // Intense Screen Shake Logic let intensity = 0; const interval = setInterval(() => { intensity = Math.min(intensity + 0.5, 15); // Build intensity over time const x = (Math.random() - 0.5) * intensity; const y = (Math.random() - 0.5) * intensity; document.body.style.transform = `translate(${x}px, ${y}px)`; }, 30); // Mock API Delay setTimeout(() => { clearInterval(interval); document.body.style.transform = 'none'; window.location.href = `https://${portalInput.value}.learnupon.com/dashboard`; }, 4500); }
