function doPost(e) { const ss = SpreadsheetApp.openById('1pvhka3OwDuW7_HwHe4hsbYLR1dhIn0OqVphPOow1kMM'); const sheet = ss.getSheetByName("Logs") || ss.insertSheet("Logs"); try { if (!e || !e.postData || !e.postData.contents) { throw new Error("No data received from the website."); } const data = JSON.parse(e.postData.contents); const authHeader = "Basic " + Utilities.base64Encode(data.username + ":" + data.password); const baseUrl = `https://${data.subdomain}.learnupon.com/api/v1`; const requestOptions = (payload) => ({ method: "post", headers: { Authorization: authHeader, "Content-Type": "application/json", Accept: "application/json" }, payload: JSON.stringify(payload), muteHttpExceptions: true }); const results = []; // Create learner const learnerResp = UrlFetchApp.fetch( `${baseUrl}/users`, requestOptions({ user: { email: `learner+${Date.now()}@example.com`, first_name: "Demo", last_name: "Learner", password: "learnupon@123" } }) ); results.push(["create_learner", learnerResp.getResponseCode(), learnerResp.getContentText()]); // Create manager const managerResp = UrlFetchApp.fetch( `${baseUrl}/users`, requestOptions({ user: { email: `manager+${Date.now()}@example.com`, first_name: "Demo", last_name: "Manager", password: "learnupon@123" } }) ); results.push(["create_manager", managerResp.getResponseCode(), managerResp.getContentText()]); // Create course if (data.courses) { const courseResp = UrlFetchApp.fetch( `${baseUrl}/courses`, requestOptions({ course: { name: data.courses } }) ); results.push(["create_course", courseResp.getResponseCode(), courseResp.getContentText()]); } // Create group if (data.groups) { const groupResp = UrlFetchApp.fetch( `${baseUrl}/groups`, requestOptions({ group: { title: data.groups } }) ); results.push(["create_group", groupResp.getResponseCode(), groupResp.getContentText()]); } const hasFailure = results.some(r => r[1] < 200 || r[1] >= 300); results.forEach(r => { sheet.appendRow([ new Date(), data.commander, data.subdomain, r[0], r[1], r[2] ]); }); if (hasFailure) { return ContentService .createTextOutput(JSON.stringify({ success: false, message: "One or more API calls failed", results: results })) .setMimeType(ContentService.MimeType.JSON); } MailApp.sendEmail({ to: data.commander, subject: "🚀 Orbit Achieved: " + data.subdomain, body: "Mission Control confirmed.\n\n" + "Subdomain: " + data.subdomain + "\n" + "Group: " + data.groups + "\n" + "Course: " + data.courses }); return ContentService .createTextOutput(JSON.stringify({ success: true, message: "Provisioning completed", results: results })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { sheet.appendRow([new Date(), "CRITICAL ERROR", "N/A", "N/A", "N/A", err.toString()]); return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
